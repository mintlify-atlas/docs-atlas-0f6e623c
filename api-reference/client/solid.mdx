## Overview

The Solid client provides SolidJS-specific signals built on top of the core authentication client. It uses SolidJS's reactive primitives for fine-grained reactivity and optimal performance.

## Installation

```bash
npm install better-auth
```

## Import

```tsx
import { createAuthClient } from "better-auth/solid"
```

## Setup

```typescript client.ts
import { createAuthClient } from "better-auth/solid"

const client = createAuthClient({
	baseURL: "http://localhost:3000"
})

export default client
```

## Signals

### useSession

SolidJS signal for accessing session data with automatic reactivity.

#### Signature

```typescript
function useSession(): Accessor<{
	data: Session | null
	error: BetterFetchError | null
	isPending: boolean
	isRefetching: boolean
	refetch: (queryParams?: { query?: SessionQueryParams }) => Promise<void>
}>
```

#### Return Type

Returns a SolidJS `Accessor` (signal getter) containing:

<ResponseField name="data" type="Session | null">
	The current session data including user and session information. `null` if not authenticated.
	
	```typescript
	type Session = {
		user: {
			id: string
			email: string
			name: string
			emailVerified: boolean
			image?: string
			createdAt: Date
			updatedAt: Date
		}
		session: {
			id: string
			userId: string
			expiresAt: Date
			token: string
			ipAddress?: string
			userAgent?: string
		}
	}
	```
</ResponseField>

<ResponseField name="error" type="BetterFetchError | null">
	Error object if the session fetch failed.
</ResponseField>

<ResponseField name="isPending" type="boolean">
	`true` during the initial session fetch, `false` once complete.
</ResponseField>

<ResponseField name="isRefetching" type="boolean">
	`true` when refetching an existing session, `false` otherwise.
</ResponseField>

<ResponseField name="refetch" type="(queryParams?: { query?: SessionQueryParams }) => Promise<void>">
	Function to manually refetch the session data.
</ResponseField>

#### Usage

```tsx
import { Show } from 'solid-js'
import client from './client'

function UserProfile() {
	const session = client.useSession()

	return (
		<Show
			when={!session().isPending}
			fallback={<div>Loading...</div>}
		>
			<Show
				when={session().error}
				fallback={
					<Show
						when={session().data}
						fallback={<div>Not logged in</div>}
					>
						<div>
							<h1>Welcome, {session().data!.user.name}</h1>
							<p>Email: {session().data!.user.email}</p>
						</div>
					</Show>
				}
			>
				<div>Error: {session().error!.message}</div>
			</Show>
		</Show>
	)
}
```

#### Manual Refetch

```tsx
import client from './client'

function SessionManager() {
	const session = client.useSession()

	const handleRefresh = async () => {
		await session().refetch({
			query: {
				disableCookieCache: true
			}
		})
	}

	return (
		<button onClick={handleRefresh}>
			Refresh Session
		</button>
	)
}
```

### Dynamic Plugin Signals

Plugins automatically generate signals based on their atoms.

```tsx
import { createAuthClient } from "better-auth/solid"
import { organizationClient } from "better-auth/client/plugins"

const client = createAuthClient({
	plugins: [organizationClient()]
})

// Auto-generated signal
function OrganizationSelector() {
	const organization = client.useActiveOrganization()

	return (
		<Show when={organization()}>
			<div>Active Org: {organization()!.name}</div>
		</Show>
	)
}
```

## Authentication Methods

All authentication methods are auto-generated based on your server configuration and are fully type-safe.

### Sign In

```tsx
import { createSignal, Show } from 'solid-js'
import client from './client'

function SignInForm() {
	const [email, setEmail] = createSignal('')
	const [password, setPassword] = createSignal('')
	const [error, setError] = createSignal<string | null>(null)

	const handleSubmit = async (e: Event) => {
		e.preventDefault()

		const result = await client.signIn.email({
			email: email(),
			password: password()
		})

		if (result.error) {
			setError(result.error.message)
			return
		}

		console.log('Signed in:', result.data)
		setError(null)
	}

	return (
		<form onSubmit={handleSubmit}>
			<Show when={error()}>
				<div class="error">{error()}</div>
			</Show>
			<input
				type="email"
				value={email()}
				onInput={(e) => setEmail(e.currentTarget.value)}
				placeholder="Email"
			/>
			<input
				type="password"
				value={password()}
				onInput={(e) => setPassword(e.currentTarget.value)}
				placeholder="Password"
			/>
			<button type="submit">Sign In</button>
		</form>
	)
}
```

## Protected Routes

### SolidStart

```tsx
import { Navigate } from '@solidjs/router'
import { Show } from 'solid-js'
import client from './client'

function ProtectedRoute(props: { children: any }) {
	const session = client.useSession()

	return (
		<Show
			when={!session().isPending}
			fallback={<div>Loading...</div>}
		>
			<Show
				when={session().data}
				fallback={<Navigate href="/login" />}
			>
				{props.children}
			</Show>
		</Show>
	)
}

// Usage
function App() {
	return (
		<Routes>
			<Route path="/login" component={LoginPage} />
			<Route
				path="/dashboard"
				component={() => (
					<ProtectedRoute>
						<Dashboard />
					</ProtectedRoute>
				)}
			/>
		</Routes>
	)
}
```

## Memos and Derived Signals

```tsx
import { createMemo } from 'solid-js'
import client from './client'

function UserInfo() {
	const session = client.useSession()

	const userName = createMemo(() => session().data?.user.name ?? 'Guest')
	const isEmailVerified = createMemo(() => session().data?.user.emailVerified ?? false)
	const isLoading = createMemo(() => session().isPending || session().isRefetching)

	return (
		<div>
			<p>Welcome, {userName()}</p>
			<Show when={isEmailVerified()}>
				<span class="badge">Verified</span>
			</Show>
			<Show when={isLoading()}>
				<span class="spinner">Loading...</span>
			</Show>
		</div>
	)
}
```

## Error Handling

```tsx
import { createSignal } from 'solid-js'
import client from './client'

function SignInWithErrorHandling() {
	const [error, setError] = createSignal<string | null>(null)

	const handleSubmit = async (email: string, password: string) => {
		const { error: apiError } = await client.signIn.email({
			email,
			password
		})

		if (apiError) {
			// Type-safe error code checking
			if (apiError.code === client.$ERROR_CODES.INVALID_EMAIL) {
				setError('Invalid email address')
			} else if (apiError.code === client.$ERROR_CODES.INVALID_PASSWORD) {
				setError('Invalid password')
			} else {
				setError('An error occurred. Please try again.')
			}
			return
		}

		setError(null)
	}

	return (
		<Show when={error()}>
			<div class="error">{error()}</div>
		</Show>
	)
}
```

## Context Provider Pattern

```tsx AuthProvider.tsx
import { createContext, useContext, ParentComponent } from 'solid-js'
import client from './client'

const AuthContext = createContext(client.useSession)

export const AuthProvider: ParentComponent = (props) => {
	const session = client.useSession()

	return (
		<AuthContext.Provider value={session}>
			{props.children}
		</AuthContext.Provider>
	)
}

export const useAuth = () => {
	const context = useContext(AuthContext)
	if (!context) {
		throw new Error('useAuth must be used within AuthProvider')
	}
	return context
}
```

```tsx App.tsx
import { AuthProvider, useAuth } from './AuthProvider'

function UserMenu() {
	const session = useAuth()

	return (
		<Show when={session().data}>
			<div>{session().data!.user.name}</div>
		</Show>
	)
}

function App() {
	return (
		<AuthProvider>
			<UserMenu />
		</AuthProvider>
	)
}
```

## useStore Helper

Low-level helper for subscribing to nanostores atoms with SolidJS.

### Signature

```typescript
function useStore<SomeStore extends Store>(
	store: SomeStore
): Accessor<StoreValue<SomeStore>>
```

### Usage

```tsx
import { useStore } from 'better-auth/solid'
import client from './client'

function SessionDebug() {
	// Access the raw session atom
	const sessionAtom = client.$store.atoms.session
	const sessionState = useStore(sessionAtom)

	return (
		<pre>
			{JSON.stringify(sessionState(), null, 2)}
		</pre>
	)
}
```

## Notes

<Note>
The Solid client uses SolidJS's fine-grained reactivity system. Only components that access specific signal values will re-render when those values change.
</Note>

<Warning>
Always call signals as functions `session()` to access their values. Forgetting the parentheses will not trigger reactivity.
</Warning>

<Info>
The Solid client automatically cleans up subscriptions when components are disposed using `onCleanup`.
</Info>
