The main function to create a Better Auth instance with full Kysely database support.

## Usage

```typescript
import { betterAuth } from "better-auth";

const auth = betterAuth({
  database: new PostgresDialect({ 
    connection: process.env.DATABASE_URL 
  }),
  // ... other options
});
```

## Parameters

<ParamField path="options" type="BetterAuthOptions" required>
  Configuration options for Better Auth. See [Configuration](/api-reference/server/configuration) for all available options.
</ParamField>

## Returns

Returns an `Auth` instance with the following properties:

<ResponseField name="handler" type="(request: Request) => Promise<Response>">
  HTTP handler function that processes authentication requests. Use this to integrate Better Auth with your web framework.

  ```typescript
  // Example with Next.js App Router
  export const { GET, POST } = auth.handler;
  ```
</ResponseField>

<ResponseField name="api" type="InferAPI">
  Server-side API methods for authentication operations. Call these directly on the server without making HTTP requests.

  <Expandable title="Available Methods">
    <ResponseField name="getSession" type="function">
      Get the current session with headers

      ```typescript
      const session = await auth.api.getSession({
        headers: request.headers
      });
      ```
    </ResponseField>

    <ResponseField name="signUpEmail" type="function">
      Create a new user account with email/password
    </ResponseField>

    <ResponseField name="signInEmail" type="function">
      Sign in with email/password
    </ResponseField>

    <ResponseField name="signInSocial" type="function">
      Initiate social provider authentication
    </ResponseField>

    <ResponseField name="signOut" type="function">
      Sign out and invalidate session
    </ResponseField>

    <ResponseField name="updateUser" type="function">
      Update user profile information
    </ResponseField>

    <ResponseField name="deleteUser" type="function">
      Delete user account
    </ResponseField>

    <ResponseField name="listSessions" type="function">
      List all active sessions for a user
    </ResponseField>

    <ResponseField name="revokeSession" type="function">
      Revoke a specific session
    </ResponseField>

    <ResponseField name="revokeSessions" type="function">
      Revoke all sessions for a user
    </ResponseField>

    <ResponseField name="revokeOtherSessions" type="function">
      Revoke all sessions except the current one
    </ResponseField>

    Additional methods are available from plugins.
  </Expandable>
</ResponseField>

<ResponseField name="options" type="BetterAuthOptions">
  The resolved configuration options passed during initialization.
</ResponseField>

<ResponseField name="$ERROR_CODES" type="ErrorCodes">
  Object containing all available error codes, including both core and plugin-specific codes.

  ```typescript
  auth.$ERROR_CODES.INVALID_EMAIL_OR_PASSWORD
  auth.$ERROR_CODES.EMAIL_NOT_VERIFIED
  ```
</ResponseField>

<ResponseField name="$context" type="Promise<AuthContext>">
  Promise that resolves to the full authentication context, including database adapters, session config, and plugin contexts.

  ```typescript
  const context = await auth.$context;
  ```
</ResponseField>

<ResponseField name="$Infer" type="InferPluginTypes">
  Type utility for inferring session and user types with plugin extensions.

  ```typescript
  type Session = typeof auth.$Infer.Session;
  ```
</ResponseField>

## Minimal Mode

For use without Kysely (e.g., with Drizzle or Prisma adapters):

```typescript
import { betterAuth } from "better-auth/minimal";
import { drizzleAdapter } from "better-auth/adapters/drizzle";

const auth = betterAuth({
  database: drizzleAdapter(db, { provider: "pg" }),
});
```

## Example

```typescript auth.ts
import { betterAuth } from "better-auth";
import { Pool } from "pg";

export const auth = betterAuth({
  database: new Pool({
    connectionString: process.env.DATABASE_URL,
  }),
  emailAndPassword: {
    enabled: true,
    requireEmailVerification: true,
  },
  socialProviders: {
    github: {
      clientId: process.env.GITHUB_CLIENT_ID!,
      clientSecret: process.env.GITHUB_CLIENT_SECRET!,
    },
  },
  session: {
    expiresIn: 60 * 60 * 24 * 7, // 7 days
    updateAge: 60 * 60 * 24, // 1 day
  },
});
```
