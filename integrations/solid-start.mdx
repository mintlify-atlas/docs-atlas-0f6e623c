Better Auth provides full support for SolidStart, enabling server-side and client-side authentication with SolidJS's fine-grained reactivity.

## Prerequisites

Make sure you have a Better Auth instance configured. If not, check out the [installation guide](/installation).

## Installation

<Steps>

### Mount the Handler

Create a catch-all API route to handle authentication requests:

```ts title="src/routes/api/auth/[...all].ts"
import { auth } from "~/lib/auth";
import { toSolidStartHandler } from "better-auth/solid-start";

export const { GET, POST } = toSolidStartHandler(auth);
```

This mounts the authentication API at `/api/auth/*` and handles all authentication requests.

<Note>
You can customize the auth path in your Better Auth config, but `/api/auth` is recommended.
</Note>

### Create Client Instance

Create a client for making authentication requests from your Solid components:

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/solid";

export const authClient = createAuthClient({
  // Optional: customize configuration
});
```

<Note type="info">
Import from `better-auth/solid` for SolidJS-specific reactivity using signals and [nanostores](https://github.com/nanostores/nanostores).
</Note>

</Steps>

## Usage

### Client-Side Components

Use the `useSession` accessor in your Solid components:

```tsx title="src/components/UserProfile.tsx"
import { authClient } from "~/lib/auth-client";
import { Show } from "solid-js";

export function UserProfile() {
  const session = authClient.useSession();

  return (
    <Show
      when={!session().isPending}
      fallback={<p>Loading...</p>}
    >
      <Show
        when={session().data}
        fallback={
          <button
            onClick={async () => {
              await authClient.signIn.social({
                provider: "github"
              });
            }}
          >
            Sign In with GitHub
          </button>
        }
      >
        {(user) => (
          <div>
            <h1>Welcome {user().user.name}!</h1>
            <p>Email: {user().user.email}</p>
            <button onClick={() => authClient.signOut()}>
              Sign Out
            </button>
          </div>
        )}
      </Show>
    </Show>
  );
}
```

### Server Functions

Access session data in server functions:

```ts title="src/lib/session.ts"
"use server";

import { auth } from "~/lib/auth";
import { getRequestEvent } from "solid-js/web";

export async function getUser() {
  const event = getRequestEvent();
  
  if (!event) {
    return null;
  }

  const session = await auth.api.getSession({
    headers: event.request.headers
  });

  return session?.user || null;
}

export async function requireAuth() {
  const user = await getUser();
  
  if (!user) {
    throw new Error("Unauthorized");
  }
  
  return user;
}
```

### Protected Routes

Create protected routes using server functions:

```tsx title="src/routes/dashboard.tsx"
import { createAsync, redirect } from "@solidjs/router";
import { auth } from "~/lib/auth";
import { getRequestEvent } from "solid-js/web";

export async function getSession() {
  "use server";
  
  const event = getRequestEvent();
  if (!event) return null;

  const session = await auth.api.getSession({
    headers: event.request.headers
  });

  if (!session) {
    throw redirect("/sign-in");
  }

  return session;
}

export default function DashboardPage() {
  const session = createAsync(() => getSession());

  return (
    <div>
      <h1>Welcome {session()?.user.name}!</h1>
      <p>Email: {session()?.user.email}</p>
    </div>
  );
}
```

### Form Actions

Handle authentication with SolidStart actions:

```tsx title="src/routes/sign-in.tsx"
import { action, useAction, useSubmission } from "@solidjs/router";
import { auth } from "~/lib/auth";
import { redirect } from "@solidjs/router";

const signInAction = action(async (formData: FormData) => {
  "use server";
  
  const email = formData.get("email") as string;
  const password = formData.get("password") as string;

  try {
    await auth.api.signInEmail({
      body: { email, password }
    });
    return redirect("/dashboard");
  } catch (error) {
    return { error: "Invalid credentials" };
  }
});

export default function SignInPage() {
  const signIn = useAction(signInAction);
  const submission = useSubmission(signInAction);

  return (
    <form action={signIn} method="post">
      {submission.result?.error && (
        <p class="error">{submission.result.error}</p>
      )}
      
      <input
        type="email"
        name="email"
        placeholder="Email"
        required
      />
      <input
        type="password"
        name="password"
        placeholder="Password"
        required
      />
      <button type="submit" disabled={submission.pending}>
        {submission.pending ? "Signing in..." : "Sign In"}
      </button>
    </form>
  );
}
```

## Example: Complete Sign In Page

<CodeGroup>

```tsx title="src/routes/sign-in.tsx (Client-Side)"
import { authClient } from "~/lib/auth-client";
import { useNavigate } from "@solidjs/router";
import { createSignal } from "solid-js";

export default function SignInPage() {
  const navigate = useNavigate();
  const [email, setEmail] = createSignal("");
  const [password, setPassword] = createSignal("");
  const [error, setError] = createSignal("");
  const [loading, setLoading] = createSignal(false);

  async function handleSignIn(e: Event) {
    e.preventDefault();
    setLoading(true);
    setError("");

    await authClient.signIn.email(
      {
        email: email(),
        password: password()
      },
      {
        onSuccess: () => {
          navigate("/dashboard");
        },
        onError: (ctx) => {
          setError(ctx.error.message);
          setLoading(false);
        }
      }
    );
  }

  return (
    <div>
      <h1>Sign In</h1>
      <form onSubmit={handleSignIn}>
        {error() && <p class="error">{error()}</p>}
        
        <input
          type="email"
          value={email()}
          onInput={(e) => setEmail(e.currentTarget.value)}
          placeholder="Email"
          required
        />
        <input
          type="password"
          value={password()}
          onInput={(e) => setPassword(e.currentTarget.value)}
          placeholder="Password"
          required
        />
        <button type="submit" disabled={loading()}>
          {loading() ? "Signing in..." : "Sign In"}
        </button>
      </form>
    </div>
  );
}
```

## API Routes

Create protected API routes:

```ts title="src/routes/api/user.ts"
import { auth } from "~/lib/auth";
import { json } from "@solidjs/router";
import type { APIEvent } from "@solidjs/start/server";

export async function GET(event: APIEvent) {
  const session = await auth.api.getSession({
    headers: event.request.headers
  });

  if (!session) {
    return json({ error: "Unauthorized" }, { status: 401 });
  }

  return json({
    user: session.user
  });
}
```

## Middleware

Protect routes using SolidStart middleware:

```ts title="src/middleware.ts"
import { createMiddleware } from "@solidjs/start/middleware";
import { auth } from "~/lib/auth";
import { redirect } from "@solidjs/router";

export default createMiddleware({
  onRequest: async (event) => {
    const protectedRoutes = ["/dashboard", "/settings"];
    const isProtectedRoute = protectedRoutes.some((route) =>
      event.request.url.includes(route)
    );

    if (isProtectedRoute) {
      const session = await auth.api.getSession({
        headers: event.request.headers
      });

      if (!session) {
        return redirect("/sign-in");
      }
    }
  }
});
```

<Note type="warn">
Always validate the session in your protected routes and server functions, not just in middleware.
</Note>
