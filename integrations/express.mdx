This guide shows you how to integrate Better Auth with [Express.js](https://expressjs.com/) to create a backend authentication server.

## Prerequisites

Make sure you have a Better Auth instance configured. If not, check out the [installation guide](/docs/installation).

<Note type="warn">
CommonJS (cjs) is not supported. Use ECMAScript Modules (ESM) by setting `"type": "module"` in your `package.json` or configure your `tsconfig.json` to use ES modules.
</Note>

## Installation

<Steps>

### Install Dependencies

```bash
npm install express better-auth
# or
pnpm add express better-auth
# or
yarn add express better-auth
```

### Mount the Handler

Create an Express server and mount the Better Auth handler to a catch-all route:

<Tabs items={['Express v4', 'Express v5']}>

<Tab value="Express v4">

```ts title="server.ts"
import express from "express";
import { toNodeHandler } from "better-auth/node";
import { auth } from "./auth";

const app = express();
const port = 3000;

// Mount Better Auth handler (catch-all for /api/auth/*)
app.all("/api/auth/*", toNodeHandler(auth));

// Mount express.json() AFTER Better Auth handler
// or only apply it to routes that don't interact with Better Auth
app.use(express.json());

// Your other routes
app.get("/", (req, res) => {
  res.send("Hello World!");
});

app.listen(port, () => {
  console.log(`Server listening on port ${port}`);
});
```

</Tab>

<Tab value="Express v5">

```ts title="server.ts"
import express from "express";
import { toNodeHandler } from "better-auth/node";
import { auth } from "./auth";

const app = express();
const port = 3000;

// Mount Better Auth handler (catch-all for /api/auth/*splat)
app.all("/api/auth/*splat", toNodeHandler(auth));

// Mount express.json() AFTER Better Auth handler
app.use(express.json());

// Your other routes
app.get("/", (req, res) => {
  res.send("Hello World!");
});

app.listen(port, () => {
  console.log(`Server listening on port ${port}`);
});
```

</Tab>

</Tabs>

<Note type="warn">
Do NOT use `express.json()` before the Better Auth handler. This will cause the client API to get stuck on "pending". Apply it only to other routes or after the Better Auth handler.
</Note>

### Verify Installation

Start your server and send a GET request to the `/ok` endpoint to verify:

```bash
curl http://localhost:3000/api/auth/ok
```

You should receive a successful response.

</Steps>

## CORS Configuration

If your frontend runs on a different origin, add CORS support:

```ts title="server.ts"
import express from "express";
import cors from "cors";
import { toNodeHandler } from "better-auth/node";
import { auth } from "./auth";

const app = express();
const port = 3000;

// Configure CORS
app.use(
  cors({
    origin: "http://localhost:5173", // Your frontend URL
    methods: ["GET", "POST", "PUT", "DELETE"],
    credentials: true, // Allow cookies
  })
);

app.all("/api/auth/*", toNodeHandler(auth));
app.use(express.json());

app.listen(port, () => {
  console.log(`Server listening on port ${port}`);
});
```

## Getting the User Session

Use the `getSession` method in your routes to retrieve the current user session:

```ts title="server.ts"
import { fromNodeHeaders } from "better-auth/node";
import { auth } from "./auth";

app.get("/api/me", async (req, res) => {
  const session = await auth.api.getSession({
    headers: fromNodeHeaders(req.headers),
  });

  if (!session) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  return res.json({
    user: session.user,
    session: session.session,
  });
});
```

The `fromNodeHeaders` helper converts Node.js request headers to the format expected by Better Auth.

## Protected Routes

Create a middleware to protect routes:

```ts title="middleware/auth.ts"
import { Request, Response, NextFunction } from "express";
import { fromNodeHeaders } from "better-auth/node";
import { auth } from "../auth";

export async function requireAuth(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const session = await auth.api.getSession({
    headers: fromNodeHeaders(req.headers),
  });

  if (!session) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  // Attach session to request object
  req.user = session.user;
  req.session = session.session;

  next();
}
```

Use the middleware:

```ts title="server.ts"
import { requireAuth } from "./middleware/auth";

app.get("/api/protected", requireAuth, (req, res) => {
  res.json({
    message: "This is a protected route",
    user: req.user,
  });
});

app.get("/api/dashboard", requireAuth, (req, res) => {
  res.json({
    user: req.user,
    data: "Dashboard data",
  });
});
```

## TypeScript Types

Extend Express Request types:

```ts title="types/express.d.ts"
import { auth } from "../auth";

type Session = typeof auth.$Infer.Session;

declare global {
  namespace Express {
    interface Request {
      user?: Session["user"];
      session?: Session["session"];
    }
  }
}
```

## Complete Example

<CodeGroup>

```ts title="auth.ts"
import { betterAuth } from "better-auth";
import { drizzleAdapter } from "better-auth/adapters/drizzle";
import { db } from "./db";

export const auth = betterAuth({
  database: drizzleAdapter(db, {
    provider: "pg",
  }),
  emailAndPassword: {
    enabled: true,
  },
  socialProviders: {
    github: {
      clientId: process.env.GITHUB_CLIENT_ID!,
      clientSecret: process.env.GITHUB_CLIENT_SECRET!,
    },
  },
});
```

```ts title="server.ts"
import express from "express";
import cors from "cors";
import { toNodeHandler, fromNodeHeaders } from "better-auth/node";
import { auth } from "./auth";
import { requireAuth } from "./middleware/auth";

const app = express();
const port = 3000;

// CORS configuration
app.use(
  cors({
    origin: "http://localhost:5173",
    credentials: true,
  })
);

// Mount Better Auth
app.all("/api/auth/*", toNodeHandler(auth));

// JSON parsing for other routes
app.use(express.json());

// Public routes
app.get("/", (req, res) => {
  res.json({ message: "Welcome to the API" });
});

app.get("/api/me", async (req, res) => {
  const session = await auth.api.getSession({
    headers: fromNodeHeaders(req.headers),
  });

  if (!session) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  res.json({ user: session.user });
});

// Protected routes
app.get("/api/dashboard", requireAuth, (req, res) => {
  res.json({
    message: "Dashboard data",
    user: req.user,
  });
});

app.post("/api/profile", requireAuth, (req, res) => {
  // Update profile logic
  res.json({
    message: "Profile updated",
    user: req.user,
  });
});

app.listen(port, () => {
  console.log(`Server running on http://localhost:${port}`);
});
```

```ts title="middleware/auth.ts"
import { Request, Response, NextFunction } from "express";
import { fromNodeHeaders } from "better-auth/node";
import { auth } from "../auth";

export async function requireAuth(
  req: Request,
  res: Response,
  next: NextFunction
) {
  try {
    const session = await auth.api.getSession({
      headers: fromNodeHeaders(req.headers),
    });

    if (!session) {
      return res.status(401).json({ error: "Unauthorized" });
    }

    req.user = session.user;
    req.session = session.session;

    next();
  } catch (error) {
    return res.status(500).json({ error: "Internal server error" });
  }
}
```

</CodeGroup>

## API Routes

Here are some common API routes you might want to implement:

### Get Current User

```ts
app.get("/api/user", requireAuth, (req, res) => {
  res.json(req.user);
});
```

### Update Profile

```ts
app.patch("/api/user", requireAuth, async (req, res) => {
  const { name } = req.body;
  
  // Update user in database
  // ...
  
  res.json({ message: "Profile updated" });
});
```

### List Sessions

```ts
app.get("/api/sessions", requireAuth, async (req, res) => {
  const sessions = await auth.api.listSessions({
    headers: fromNodeHeaders(req.headers),
  });
  
  res.json(sessions);
});
```
