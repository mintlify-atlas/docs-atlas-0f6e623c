Better Auth provides seamless integration with Nuxt, supporting both client-side and server-side authentication with Vue 3 and Nitro.

## Prerequisites

Make sure you have a Better Auth instance configured. If not, check out the [installation guide](/installation).

## Installation

<Steps>

### Mount the Handler

Create a catch-all API route to handle authentication requests:

```ts title="server/api/auth/[...].ts"
import { auth } from "~/server/lib/auth";

export default defineEventHandler(async (event) => {
  return auth.handler(toWebRequest(event));
});
```

This mounts the authentication API at `/api/auth/*` and handles all authentication requests.

<Note>
You can customize the auth path in your Better Auth config, but `/api/auth` is recommended.
</Note>

### Create Auth Instance

Create your Better Auth configuration:

```ts title="server/lib/auth.ts"
import { betterAuth } from "better-auth";
import Database from "better-sqlite3";

export const auth = betterAuth({
  database: new Database("./sqlite.db"),
  emailAndPassword: {
    enabled: true
  },
  socialProviders: {
    github: {
      clientId: process.env.GITHUB_CLIENT_ID!,
      clientSecret: process.env.GITHUB_CLIENT_SECRET!
    },
    google: {
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!
    }
  }
});
```

### Create Client Instance

Create a client for making authentication requests:

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/vue";

export const authClient = createAuthClient({
  baseURL: process.env.NUXT_PUBLIC_APP_URL
});

export const { useSession } = authClient;
```

<Note type="info">
Import from `better-auth/vue` for Vue-specific composition API using [nanostores](https://github.com/nanostores/nanostores).
</Note>

</Steps>

## Usage

### Client-Side Components

Use the `useSession` composable in your Vue components:

```vue title="components/UserProfile.vue"
<script setup lang="ts">
import { useSession } from "~/lib/auth-client";
import { authClient } from "~/lib/auth-client";

const session = useSession();

const handleSignOut = async () => {
  await authClient.signOut();
};
</script>

<template>
  <div v-if="session.isPending">
    Loading...
  </div>
  <div v-else-if="session.data">
    <h1>Welcome {{ session.data.user.name }}!</h1>
    <p>Email: {{ session.data.user.email }}</p>
    <button @click="handleSignOut">Sign Out</button>
  </div>
  <div v-else>
    <button @click="authClient.signIn.social({ provider: 'github' })">
      Sign In with GitHub
    </button>
  </div>
</template>
```

### Server Routes & API

Access session data in server API routes:

```ts title="server/api/user.ts"
import { auth } from "~/server/lib/auth";

export default defineEventHandler(async (event) => {
  const session = await auth.api.getSession({
    headers: event.headers
  });

  if (!session) {
    throw createError({
      statusCode: 401,
      message: "Unauthorized"
    });
  }

  return {
    user: session.user
  };
});
```

### Server Middleware

Protect routes using server middleware:

```ts title="server/middleware/auth.ts"
import { auth } from "~/server/lib/auth";

export default defineEventHandler(async (event) => {
  const protectedRoutes = ["/dashboard", "/settings"];
  const isProtectedRoute = protectedRoutes.some((route) =>
    event.path.startsWith(route)
  );

  if (isProtectedRoute) {
    const session = await auth.api.getSession({
      headers: event.headers
    });

    if (!session) {
      throw createError({
        statusCode: 401,
        message: "Unauthorized"
      });
    }

    // Attach session to event context
    event.context.user = session.user;
    event.context.session = session.session;
  }
});
```

### Page Middleware

Create client-side page middleware for protected routes:

```ts title="middleware/auth.ts"
import { authClient } from "~/lib/auth-client";

export default defineNuxtRouteMiddleware(async (to, from) => {
  const session = await authClient.getSession();

  if (!session.data) {
    return navigateTo("/sign-in");
  }
});
```

Use it in your pages:

```vue title="pages/dashboard.vue"
<script setup lang="ts">
definePageMeta({
  middleware: "auth"
});

import { useSession } from "~/lib/auth-client";

const session = useSession();
</script>

<template>
  <div>
    <h1>Welcome {{ session.data?.user.name }}!</h1>
  </div>
</template>
```

## Example: Sign In Page

<CodeGroup>

```vue title="pages/sign-in.vue (Client-Side)"
<script setup lang="ts">
import { ref } from "vue";
import { authClient } from "~/lib/auth-client";

const email = ref("");
const password = ref("");
const error = ref("");
const loading = ref(false);

const handleSignIn = async () => {
  loading.value = true;
  error.value = "";
  
  await authClient.signIn.email(
    {
      email: email.value,
      password: password.value
    },
    {
      onSuccess: () => {
        navigateTo("/dashboard");
      },
      onError: (ctx) => {
        error.value = ctx.error.message;
        loading.value = false;
      }
    }
  );
};
</script>

<template>
  <div>
    <h1>Sign In</h1>
    <form @submit.prevent="handleSignIn">
      <p v-if="error" style="color: red;">{{ error }}</p>
      
      <input
        v-model="email"
        type="email"
        placeholder="Email"
        required
      />
      <input
        v-model="password"
        type="password"
        placeholder="Password"
        required
      />
      <button type="submit" :disabled="loading">
        {{ loading ? "Signing in..." : "Sign In" }}
      </button>
    </form>
  </div>
</template>
```

```ts title="server/api/auth/sign-in.post.ts (Server Action)"
import { auth } from "~/server/lib/auth";

export default defineEventHandler(async (event) => {
  const { email, password } = await readBody(event);

  try {
    const session = await auth.api.signInEmail({
      body: { email, password },
      headers: event.headers
    });

    return {
      success: true,
      session
    };
  } catch (error) {
    throw createError({
      statusCode: 401,
      message: "Invalid credentials"
    });
  }
});
```

</CodeGroup>

## Composables

Create reusable composables for common auth operations:

```ts title="composables/useAuth.ts"
import { authClient } from "~/lib/auth-client";

export const useAuth = () => {
  const session = authClient.useSession();

  const signIn = async (email: string, password: string) => {
    await authClient.signIn.email({ email, password });
  };

  const signOut = async () => {
    await authClient.signOut();
  };

  const isAuthenticated = computed(() => !!session.value.data);
  const user = computed(() => session.value.data?.user);

  return {
    session,
    user,
    isAuthenticated,
    signIn,
    signOut
  };
};
```

Use it in your components:

```vue
<script setup lang="ts">
const { user, isAuthenticated, signOut } = useAuth();
</script>

<template>
  <div v-if="isAuthenticated">
    <p>Welcome {{ user?.name }}!</p>
    <button @click="signOut">Sign Out</button>
  </div>
</template>
```

## Social Sign In

```vue title="components/SocialSignIn.vue"
<script setup lang="ts">
import { authClient } from "~/lib/auth-client";

const signInWithGitHub = () => {
  authClient.signIn.social({
    provider: "github",
    callbackURL: "/dashboard"
  });
};

const signInWithGoogle = () => {
  authClient.signIn.social({
    provider: "google",
    callbackURL: "/dashboard"
  });
};
</script>

<template>
  <div>
    <button @click="signInWithGitHub">
      Continue with GitHub
    </button>
    <button @click="signInWithGoogle">
      Continue with Google
    </button>
  </div>
</template>
```

## TypeScript Types

Add types for event context:

```ts title="types/auth.d.ts"
import type { auth } from "~/server/lib/auth";

type Session = typeof auth.$Infer.Session;

declare module "h3" {
  interface H3EventContext {
    user?: Session["user"];
    session?: Session["session"];
  }
}
```
