This guide covers integrating Better Auth with standalone Vue.js applications. For Nuxt integration, see the [Nuxt integration guide](/docs/integrations/nuxt).

## Prerequisites

- A Better Auth backend instance running (see [installation guide](/docs/installation))
- A Vue application (using Vite, Vue CLI, or another setup)

## Installation

<Steps>

### Setup Backend

First, ensure you have a Better Auth backend running. This could be a separate Express server, Nuxt API routes, or any other Node.js server.

See the [Express integration](/docs/integrations/express) or other server integrations for backend setup.

### Create Client Instance

Install Better Auth and create a client instance:

```bash
npm install better-auth
# or
pnpm add better-auth
# or
yarn add better-auth
```

Create a client:

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/vue";

export const authClient = createAuthClient({
  baseURL: "http://localhost:3000", // Your backend URL
});
```

<Note type="info">
Import from `better-auth/vue` for Vue-specific reactivity using [nanostores](https://github.com/nanostores/nanostores) and Vue's Composition API.
</Note>

</Steps>

## Usage

### Session Management

Use the `useSession` composable to access session state:

```vue title="components/UserProfile.vue"
<script setup lang="ts">
import { authClient } from "@/lib/auth-client";

const session = authClient.useSession();

const handleSignOut = async () => {
  await authClient.signOut();
};
</script>

<template>
  <div v-if="session.isPending">Loading...</div>
  
  <div v-else-if="session.error">
    Error: {{ session.error.message }}
  </div>
  
  <div v-else-if="!session.data">
    Not authenticated
  </div>
  
  <div v-else>
    <h1>Welcome {{ session.data.user.name }}!</h1>
    <p>Email: {{ session.data.user.email }}</p>
    <button @click="handleSignOut">Sign Out</button>
  </div>
</template>
```

### Sign Up

```vue title="pages/SignUp.vue"
<script setup lang="ts">
import { authClient } from "@/lib/auth-client";
import { ref } from "vue";
import { useRouter } from "vue-router";

const router = useRouter();
const email = ref("");
const password = ref("");
const name = ref("");
const error = ref("");
const loading = ref(false);

const handleSignUp = async () => {
  error.value = "";
  loading.value = true;

  await authClient.signUp.email(
    {
      email: email.value,
      password: password.value,
      name: name.value
    },
    {
      onSuccess: () => {
        router.push("/dashboard");
      },
      onError: (ctx) => {
        error.value = ctx.error.message;
        loading.value = false;
      }
    }
  );
};
</script>

<template>
  <div>
    <h1>Sign Up</h1>
    <form @submit.prevent="handleSignUp">
      <p v-if="error" class="error">{{ error }}</p>
      
      <input
        v-model="name"
        type="text"
        placeholder="Name"
        required
      />
      <input
        v-model="email"
        type="email"
        placeholder="Email"
        required
      />
      <input
        v-model="password"
        type="password"
        placeholder="Password"
        required
      />
      <button type="submit" :disabled="loading">
        {{ loading ? "Creating account..." : "Sign Up" }}
      </button>
    </form>
  </div>
</template>
```

### Protected Routes

Create a navigation guard to protect routes:

```ts title="router/index.ts"
import { createRouter, createWebHistory } from "vue-router";
import { authClient } from "@/lib/auth-client";
import SignIn from "@/pages/SignIn.vue";
import Dashboard from "@/pages/Dashboard.vue";

const router = createRouter({
  history: createWebHistory(),
  routes: [
    {
      path: "/sign-in",
      component: SignIn,
      meta: { requiresAuth: false }
    },
    {
      path: "/dashboard",
      component: Dashboard,
      meta: { requiresAuth: true }
    }
  ]
});

router.beforeEach(async (to, from, next) => {
  if (to.meta.requiresAuth) {
    const session = authClient.useSession();
    
    // Wait for session to load
    if (session.value.isPending) {
      await new Promise(resolve => {
        const unwatch = watch(
          () => session.value.isPending,
          (isPending) => {
            if (!isPending) {
              unwatch();
              resolve(true);
            }
          }
        );
      });
    }
    
    if (!session.value.data) {
      next("/sign-in");
      return;
    }
  }
  
  next();
});

export default router;
```

Or create a composable for route protection:

```ts title="composables/useAuth.ts"
import { authClient } from "@/lib/auth-client";
import { useRouter } from "vue-router";
import { computed } from "vue";

export function useAuth() {
  const router = useRouter();
  const session = authClient.useSession();

  const isAuthenticated = computed(() => !!session.value.data);
  const user = computed(() => session.value.data?.user);

  const requireAuth = () => {
    if (!isAuthenticated.value && !session.value.isPending) {
      router.push("/sign-in");
    }
  };

  return {
    session,
    isAuthenticated,
    user,
    requireAuth,
    signOut: authClient.signOut
  };
}
```

Use in components:

```vue title="pages/Dashboard.vue"
<script setup lang="ts">
import { useAuth } from "@/composables/useAuth";
import { onMounted } from "vue";

const { user, requireAuth, signOut } = useAuth();

onMounted(() => {
  requireAuth();
});
</script>

<template>
  <div v-if="user">
    <h1>Welcome {{ user.name }}!</h1>
    <button @click="signOut">Sign Out</button>
  </div>
</template>
```

## API Calls with Authentication

Make authenticated API calls:

```ts title="lib/api.ts"
import { authClient } from "./auth-client";

export async function fetchUserData() {
  const { data, error } = await authClient.$fetch("/api/user", {
    method: "GET"
  });

  if (error) {
    throw new Error(error.message);
  }

  return data;
}

export async function updateProfile(name: string) {
  const { data, error } = await authClient.$fetch("/api/user", {
    method: "PATCH",
    body: { name }
  });

  if (error) {
    throw new Error(error.message);
  }

  return data;
}
```

Use in components:

```vue title="components/UserSettings.vue"
<script setup lang="ts">
import { ref } from "vue";
import { updateProfile } from "@/lib/api";
import { authClient } from "@/lib/auth-client";

const session = authClient.useSession();
const name = ref(session.value.data?.user.name || "");
const loading = ref(false);

const handleUpdate = async () => {
  loading.value = true;
  try {
    await updateProfile(name.value);
    // Refresh session to get updated data
    await session.value.refetch();
  } catch (error) {
    console.error(error);
  } finally {
    loading.value = false;
  }
};
</script>

<template>
  <form @submit.prevent="handleUpdate">
    <input v-model="name" type="text" placeholder="Name" />
    <button type="submit" :disabled="loading">
      {{ loading ? "Updating..." : "Update Profile" }}
    </button>
  </form>
</template>
```

## Session Refresh

Manually refresh the session:

```vue title="components/RefreshButton.vue"
<script setup lang="ts">
import { authClient } from "@/lib/auth-client";

const session = authClient.useSession();

const handleRefresh = async () => {
  await session.value.refetch();
};
</script>

<template>
  <button
    @click="handleRefresh"
    :disabled="session.isRefetching"
  >
    {{ session.isRefetching ? "Refreshing..." : "Refresh Session" }}
  </button>
</template>
```

## Environment Variables

Set the backend URL via environment variables:

```env title=".env"
VITE_AUTH_URL=http://localhost:3000
```

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/vue";

export const authClient = createAuthClient({
  baseURL: import.meta.env.VITE_AUTH_URL || "http://localhost:3000"
});
```

## CORS Configuration

Ensure your backend allows CORS requests from your Vue app:

```ts title="server.ts"
import express from "express";
import cors from "cors";
import { toNodeHandler } from "better-auth/node";
import { auth } from "./auth";

const app = express();

app.use(
  cors({
    origin: "http://localhost:5173", // Your Vue app URL
    credentials: true
  })
);

app.all("/api/auth/*", toNodeHandler(auth));

app.listen(3000);
```

## TypeScript Types

Infer session types from your auth instance:

```ts title="types/auth.ts"
import type { authClient } from "@/lib/auth-client";

export type Session = typeof authClient.$Infer.Session;
export type User = Session["user"];
```

Use in your components:

```vue
<script setup lang="ts">
import type { User } from "@/types/auth";

interface Props {
  user: User;
}

defineProps<Props>();
</script>

<template>
  <div>
    <h2>{{ user.name }}</h2>
    <p>{{ user.email }}</p>
  </div>
</template>
```
