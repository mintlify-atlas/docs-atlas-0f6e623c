# React Client

The React client provides a hook-based API for authentication in React applications. It's built on top of the vanilla client and provides seamless integration with React's component model.

## Installation

```bash
npm install better-auth
# or
pnpm add better-auth
# or
yarn add better-auth
```

## Basic Setup

<Steps>

### Create Client Instance

Create a client instance by importing from `better-auth/react`:

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/react";

export const authClient = createAuthClient({
  baseURL: "http://localhost:3000", // Your backend URL
});

export const { 
  useSession, 
  signIn, 
  signOut, 
  signUp 
} = authClient;
```

### Configure Options

```ts title="lib/auth-client.ts"
import { createAuthClient } from "better-auth/react";

export const authClient = createAuthClient({
  baseURL: "http://localhost:3000",
  basePath: "/api/auth",
  fetchOptions: {
    onError(ctx) {
      console.error("Auth error:", ctx.error);
    },
  },
});
```

</Steps>

## Hooks API

### useSession

The primary hook for accessing session state in React components:

```tsx title="components/UserProfile.tsx"
import { authClient } from "@/lib/auth-client";

export function UserProfile() {
  const session = authClient.useSession();

  if (session.isPending) {
    return <div>Loading...</div>;
  }

  if (session.error) {
    return <div>Error: {session.error.message}</div>;
  }

  if (!session.data) {
    return <div>Not authenticated</div>;
  }

  return (
    <div>
      <h1>Welcome {session.data.user.name}!</h1>
      <p>Email: {session.data.user.email}</p>
      <button onClick={() => authClient.signOut()}>
        Sign Out
      </button>
    </div>
  );
}
```

#### Session Object

```ts
interface UseSessionReturn {
  data: {
    session: Session | null;
    user: User | null;
  } | null;
  isPending: boolean;      // Initial loading state
  isRefetching: boolean;   // Refetching state
  error: BetterFetchError | null;
  refetch: (params?: { query?: SessionQueryParams }) => Promise<void>;
}
```

### Manual Refetch

```tsx
function SessionManager() {
  const session = authClient.useSession();

  const handleRefresh = async () => {
    await session.refetch();
  };

  const handleRefreshWithoutCache = async () => {
    await session.refetch({
      query: { disableCookieCache: true }
    });
  };

  return (
    <div>
      <button onClick={handleRefresh}>Refresh Session</button>
      <button onClick={handleRefreshWithoutCache}>
        Force Refresh
      </button>
    </div>
  );
}
```

## Authentication Methods

All authentication methods work the same as the vanilla client:

### Sign Up

```tsx title="pages/SignUp.tsx"
import { authClient } from "@/lib/auth-client";
import { useState } from "react";
import { useNavigate } from "react-router-dom";

export function SignUpPage() {
  const navigate = useNavigate();
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [name, setName] = useState("");
  const [error, setError] = useState("");

  const handleSignUp = async (e: React.FormEvent) => {
    e.preventDefault();
    setError("");

    const { data, error: signUpError } = await authClient.signUp.email({
      email,
      password,
      name,
    });

    if (signUpError) {
      setError(signUpError.message ?? "Failed to sign up");
      return;
    }

    navigate("/dashboard");
  };

  return (
    <form onSubmit={handleSignUp}>
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Name"
        required
      />
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
        required
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
        required
      />
      {error && <div className="error">{error}</div>}
      <button type="submit">Sign Up</button>
    </form>
  );
}
```

### Social Sign In

```tsx
function SocialSignIn() {
  const handleGoogleSignIn = async () => {
    await authClient.signIn.social({
      provider: "google",
      callbackURL: "/dashboard",
    });
  };

      provider: "github",
      callbackURL: "/dashboard",
```

## Protected Routes

### Route Guard Component

```tsx title="components/ProtectedRoute.tsx"
import { authClient } from "@/lib/auth-client";
import { Navigate } from "react-router-dom";
import type { ReactNode } from "react";

interface ProtectedRouteProps {
  children: ReactNode;
}

export function ProtectedRoute({ children }: ProtectedRouteProps) {
  const session = authClient.useSession();

  if (session.isPending) {
    return <div>Loading...</div>;
  }

  if (!session.data) {
    return <Navigate to="/sign-in" replace />;
  }

  return <>{children}</>;
}
```

### Usage with React Router

```tsx title="App.tsx"
import { BrowserRouter, Routes, Route } from "react-router-dom";
import { ProtectedRoute } from "./components/ProtectedRoute";
import { Dashboard } from "./pages/Dashboard";
import { SignIn } from "./pages/SignIn";

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/sign-in" element={<SignIn />} />
        <Route
          path="/dashboard"
          element={
            <ProtectedRoute>
              <Dashboard />
            </ProtectedRoute>
          }
        />
      </Routes>
    </BrowserRouter>
  );
}
```

## User Management

### Update User Profile

```tsx
function ProfileEditor() {
  const session = authClient.useSession();
  const [name, setName] = useState(session.data?.user.name ?? "");

  const handleUpdate = async (e: React.FormEvent) => {
    e.preventDefault();
    
    const { error } = await authClient.updateUser({ name });
    
    if (!error) {
      alert("Profile updated!");
    }
  };

  return (
    <form onSubmit={handleUpdate}>
      <input
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Name"
      />
      <button type="submit">Update Profile</button>
    </form>
  );
}
```

## Plugin Hooks

When using plugins, additional hooks become available:

```tsx
import { createAuthClient } from "better-auth/react";
import { organizationClient } from "better-auth/client/plugins";

const authClient = createAuthClient({
  baseURL: "http://localhost:3000",
  plugins: [organizationClient()],
});

// Plugin hooks are automatically available
function OrganizationList() {
  const { data: orgs } = authClient.useListOrganizations();

  return (
    <ul>
      {orgs?.map((org) => (
        <li key={org.id}>{org.name}</li>
      ))}
    </ul>
  );
}
```

## TypeScript Support

The React client provides full TypeScript support:

```tsx
import type { AuthClient } from "better-auth/react";

// Infer session type
type Session = typeof authClient.$Infer.Session;

function useUserRole() {
  const session = authClient.useSession();
  
  const role: string | undefined = session.data?.user.role;
  
  return role;
}
```

## Best Practices

<Note>
**Performance Tips:**
- The `useSession` hook automatically subscribes to session changes
- Session state is shared across all components using the hook
- Avoid calling authentication methods inside render functions
- Use the `isRefetching` state to show loading indicators during refetch
</Note>

<Note>
**Error Handling:**
- Always check for errors in authentication responses
- Display user-friendly error messages
- Consider implementing error boundaries for auth errors
</Note>
