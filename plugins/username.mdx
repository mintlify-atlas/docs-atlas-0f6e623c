title: Username

The username plugin adds username support to email and password authentication, allowing users to sign in with their username instead of email.

## Installation

<Steps>
  <Step>
    ### Add the plugin to the server

    ```ts title="auth.ts"
    import { betterAuth } from "better-auth"
    import { username } from "better-auth/plugins"

    export const auth = betterAuth({
      emailAndPassword: {
        enabled: true,
      },
      plugins: [
        username()
      ]
    })
    ```
  </Step>
  
  <Step>
    ### Migrate the database

    <Tabs items={["migrate", "generate"]}>
      <Tab value="migrate">
      ```bash
      npx auth migrate
      ```
      </Tab>
      <Tab value="generate">
      ```bash
      npx auth generate
      ```
      </Tab>
    </Tabs>
  </Step>
  
  <Step>
    ### Add the client plugin

    ```ts title="auth-client.ts"
    import { createAuthClient } from "better-auth/client"
    import { usernameClient } from "better-auth/client/plugins"

    export const authClient = createAuthClient({
      plugins: [
        usernameClient()
      ]
    })
    ```
  </Step>
</Steps>

## Usage

### Sign Up

Sign up with username using the existing `signUp.email` function:

```ts
import { authClient } from "@/lib/auth-client"

const { data } = await authClient.signUp.email({
  email: "user@example.com",
  name: "John Doe",
  password: "secure-password",
  username: "johndoe",
  displayUsername: "JohnDoe123" // Optional
})
```

<Callout type="info">
If only `username` is provided, `displayUsername` will be set to the pre-normalized version of `username`.
</Callout>

### Sign In

Sign in with username:

```ts
import { authClient } from "@/lib/auth-client"

const { data } = await authClient.signIn.username({
  username: "johndoe",
  password: "secure-password"
})
```

### Update Username

Update a user's username:

```ts
import { authClient } from "@/lib/auth-client"

const { data } = await authClient.updateUser({
  username: "new-username"
})
```

### Check Username Availability

Check if a username is available:

```ts
import { authClient } from "@/lib/auth-client"

const { data } = await authClient.isUsernameAvailable({
  username: "johndoe"
})

if (data?.available) {
  console.log("Username is available")
} else {
  console.log("Username is taken")
}
```

## Options

```ts title="auth.ts"
import { betterAuth } from "better-auth"
import { username } from "better-auth/plugins"

export const auth = betterAuth({
  emailAndPassword: {
    enabled: true,
  },
  plugins: [
    username({
      // Minimum username length
      minUsernameLength: 3, // Default: 3
      
      // Maximum username length
      maxUsernameLength: 30, // Default: 30
      
      // Custom username validator
      usernameValidator: (username) => {
        // Return false if invalid
        if (username === "admin") {
          return false
        }
        return true
      },
      
      // Custom display username validator
      displayUsernameValidator: (displayUsername) => {
        // Allow only alphanumeric, underscores, and hyphens
        return /^[a-zA-Z0-9_-]+$/.test(displayUsername)
      },
      
      // Username normalization function
      usernameNormalization: (username) => {
        return username.toLowerCase()
          .replaceAll("0", "o")
          .replaceAll("3", "e")
          .replaceAll("4", "a")
      },
      
      // Or disable normalization
      usernameNormalization: false,
      
      // Display username normalization
      displayUsernameNormalization: (displayUsername) => {
        return displayUsername.toLowerCase()
      },
      
      // Or disable display username normalization
      displayUsernameNormalization: false,
      
      // Validation order
      validationOrder: {
        username: "post-normalization", // or "pre-normalization" (default)
        displayUsername: "post-normalization",
      },
    })
  ]
})
```

### Username Normalization

By default, usernames are normalized to lowercase, so "TestUser" and "testuser" are considered the same:

- The `username` field stores the normalized (lowercase) username
- The `displayUsername` field stores the original username

You can customize or disable normalization:

```ts
// Custom normalization
usernameNormalization: (username) => {
  return username.toLowerCase().replaceAll("0", "o")
}

// Disable normalization
usernameNormalization: false
```

### Display Username

The `displayUsername` field allows users to have a presentable version of their username:

- When only `username` is provided during signup, `displayUsername` is set to the original value before normalization
- Users can explicitly set a `displayUsername` which will be preserved as-is
- By default, display usernames are not normalized

### Disable Username Availability Endpoint

To prevent username enumeration:

```ts title="auth.ts"
import { betterAuth } from "better-auth"
import { username } from "better-auth/plugins"

export const auth = betterAuth({
  emailAndPassword: {
    enabled: true,
  },
  disabledPaths: ["/is-username-available"],
  plugins: [
    username()
  ]
})
```

## Schema

The plugin adds these fields to the user table:

- `username` (string, unique): Normalized username
- `displayUsername` (string, unique): Non-normalized username for display

## Example: Complete Implementation

```tsx title="app/signup/page.tsx"
import { authClient } from "@/lib/auth-client"
import { useState, useEffect } from "react"

export default function SignupPage() {
  const [formData, setFormData] = useState({
    email: "",
    name: "",
    username: "",
    password: "",
  })
  const [usernameAvailable, setUsernameAvailable] = useState<boolean | null>(null)
  const [checking, setChecking] = useState(false)
  const [error, setError] = useState("")
  
  // Check username availability with debounce
  useEffect(() => {
    if (formData.username.length < 3) {
      setUsernameAvailable(null)
      return
    }
    
    const timer = setTimeout(async () => {
      setChecking(true)
      const { data } = await authClient.isUsernameAvailable({
        username: formData.username
      })
      setUsernameAvailable(data?.available ?? false)
      setChecking(false)
    }, 500)
    
    return () => clearTimeout(timer)
  }, [formData.username])
  
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setError("")
    
    if (!usernameAvailable) {
      setError("Username is not available")
      return
    }
    
    const { data, error: signUpError } = await authClient.signUp.email(formData)
    
    if (signUpError) {
      setError(signUpError.message)
    } else {
      // Redirect to dashboard
      window.location.href = "/dashboard"
    }
  }
  
  return (
    <form onSubmit={handleSubmit}>
      <h2>Sign Up</h2>
      
      <input
        type="email"
        value={formData.email}
        onChange={(e) => setFormData({ ...formData, email: e.target.value })}
        placeholder="Email"
        required
      />
      
      <input
        type="text"
        value={formData.name}
        onChange={(e) => setFormData({ ...formData, name: e.target.value })}
        placeholder="Name"
        required
      />
      
      <div>
        <input
          type="text"
          value={formData.username}
          onChange={(e) => setFormData({ ...formData, username: e.target.value })}
          placeholder="Username"
          required
          minLength={3}
          maxLength={30}
        />
        {checking && <span>Checking...</span>}
        {usernameAvailable === true && <span>✓ Available</span>}
        {usernameAvailable === false && <span>✗ Not available</span>}
      </div>
      
      <input
        type="password"
        value={formData.password}
        onChange={(e) => setFormData({ ...formData, password: e.target.value })}
        placeholder="Password"
        required
      />
      
      <button type="submit" disabled={!usernameAvailable}>
        Sign Up
      </button>
      
      {error && <p>{error}</p>}
    </form>
  )
}
```

```tsx title="app/login/page.tsx"
import { authClient } from "@/lib/auth-client"
import { useState } from "react"

export default function LoginPage() {
  const [username, setUsername] = useState("")
  const [password, setPassword] = useState("")
  const [error, setError] = useState("")
  
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setError("")
    
    const { data, error: signInError } = await authClient.signIn.username({
      username,
      password,
    })
    
    if (signInError) {
      setError(signInError.message)
    } else {
      window.location.href = "/dashboard"
    }
  }
  
  return (
    <form onSubmit={handleSubmit}>
      <h2>Sign In</h2>
      
      <input
        type="text"
        value={username}
        onChange={(e) => setUsername(e.target.value)}
        placeholder="Username"
        required
      />
      
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
        required
      />
      
      <button type="submit">Sign In</button>
      
      {error && <p>{error}</p>}
    </form>
  )
}
```
