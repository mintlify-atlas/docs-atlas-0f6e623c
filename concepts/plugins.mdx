## Overview

Better Auth's plugin system allows you to extend functionality in a modular way. Plugins can add endpoints, database schemas, middleware, hooks, and more.

## Using Plugins

### Basic Usage

Import and add plugins to your configuration:

```typescript
import { betterAuth } from "better-auth";
import { twoFactor, organization, admin } from "better-auth/plugins";

const auth = betterAuth({
  plugins: [
    twoFactor(),
    organization(),
    admin()
  ]
});
```

### Plugin Configuration

Most plugins accept configuration options:

```typescript
import { twoFactor } from "better-auth/plugins";

const auth = betterAuth({
  plugins: [
    twoFactor({
      issuer: "MyApp",
      totpWindow: 2,
      backupCodeLength: 10,
      backupCodeCount: 10
    })
  ]
});
```

## Plugin Structure

A plugin is an object that conforms to the `BetterAuthPlugin` interface:

```typescript
import type { BetterAuthPlugin } from "better-auth";

export const myPlugin = (options?: MyPluginOptions) => {
  return {
    id: "my-plugin",
    
    // Optional: Add endpoints
    endpoints: {
      myEndpoint: createAuthEndpoint(
        "/my-endpoint",
        { method: "POST" },
        async (ctx) => {
          // Handle request
          return ctx.json({ success: true });
        }
      )
    },
    
    // Optional: Add database schema
    schema: {
      myTable: {
        fields: {
          id: { type: "string", required: true },
          userId: { type: "string", required: true, references: { model: "user", field: "id" } },
          data: { type: "string" },
          createdAt: { type: "date", required: true }
        }
      }
    },
    
    // Optional: Initialize plugin
    async init(ctx) {
      // Setup logic
      return {
        context: {
          // Add to context
          myPluginData: "value"
        }
      };
    },
    
    // Optional: Add hooks
    hooks: {
      after: [
        {
          matcher: (ctx) => ctx.path === "/sign-in",
          handler: createAuthMiddleware(async (ctx) => {
            // Run after sign-in
          })
        }
      ]
    },
    
    // Optional: Plugin options
    options,
    
    // Optional: Error codes
    $ERROR_CODES: {
      MY_ERROR: {
        code: "MY_ERROR",
        message: "Something went wrong"
      }
    }
  } satisfies BetterAuthPlugin;
};
```

## Creating a Plugin

Let's create a complete plugin example:

<Steps>

### Step 1: Define Plugin Options

```typescript
interface RateLimitPluginOptions {
  maxAttempts?: number;
  windowMs?: number;
}
```

### Step 2: Create Database Schema

```typescript
const schema = {
  rateLimitLog: {
    fields: {
      id: { type: "string", required: true },
      identifier: { type: "string", required: true },
      attempts: { type: "number", required: true },
      windowStart: { type: "date", required: true },
      createdAt: { type: "date", required: true }
    }
  }
} satisfies BetterAuthPluginDBSchema;
```

### Step 3: Create Endpoints

```typescript
import { createAuthEndpoint } from "better-auth";
import * as z from "zod";

const endpoints = {
  checkRateLimit: createAuthEndpoint(
    "/check-rate-limit",
    {
      method: "POST",
      body: z.object({
        identifier: z.string()
      })
    },
    async (ctx) => {
      const { identifier } = ctx.body;
      
      // Check rate limit
      const log = await ctx.context.adapter.findOne({
        model: "rateLimitLog",
        where: [{ field: "identifier", value: identifier }]
      });
      
      if (log && log.attempts >= ctx.context.options.maxAttempts) {
        throw APIError.from("TOO_MANY_REQUESTS", {
          message: "Rate limit exceeded"
        });
      }
      
      return ctx.json({ allowed: true });
    }
  )
};
```

### Step 4: Add Hooks

```typescript
const hooks = {
  before: [
    {
      matcher: (ctx) => ctx.path?.startsWith("/sign-in"),
      handler: createAuthMiddleware(async (ctx) => {
        // Check rate limit before sign-in
        const ip = getIp(ctx.request);
        
        const log = await ctx.context.adapter.findOne({
          model: "rateLimitLog",
          where: [{ field: "identifier", value: ip }]
        });
        
        if (log && log.attempts >= options.maxAttempts) {
          throw APIError.from("TOO_MANY_REQUESTS", {
            message: "Too many sign-in attempts"
          });
        }
      })
    }
  ],
  after: [
    {
      matcher: (ctx) => ctx.path?.startsWith("/sign-in"),
      handler: createAuthMiddleware(async (ctx) => {
        // Log attempt after sign-in
        const ip = getIp(ctx.request);
        
        await ctx.context.adapter.create({
          model: "rateLimitLog",
          data: {
            identifier: ip,
            attempts: 1,
            windowStart: new Date(),
            createdAt: new Date()
          }
        });
      })
    }
  ]
};
```

### Step 5: Combine Into Plugin

```typescript
export const rateLimit = (options?: RateLimitPluginOptions) => {
  const defaultOptions = {
    maxAttempts: 5,
    windowMs: 15 * 60 * 1000, // 15 minutes
    ...options
  };
  
  return {
    id: "rate-limit",
    endpoints,
    schema,
    hooks,
    options: defaultOptions,
    $ERROR_CODES: {
      RATE_LIMIT_EXCEEDED: {
        code: "RATE_LIMIT_EXCEEDED",
        message: "Rate limit exceeded. Please try again later."
      }
    }
  } satisfies BetterAuthPlugin;
};
```

### Step 6: Use Your Plugin

```typescript
import { betterAuth } from "better-auth";
import { rateLimit } from "./rate-limit-plugin";

const auth = betterAuth({
  plugins: [
    rateLimit({
      maxAttempts: 3,
      windowMs: 10 * 60 * 1000 // 10 minutes
    })
  ]
});
```

</Steps>

## Plugin Initialization

The `init` function runs when the auth instance is created:

```typescript
const myPlugin = () => {
  return {
    id: "my-plugin",
    async init(ctx) {
      // Access context
      console.log("Base URL:", ctx.baseURL);
      console.log("Database:", ctx.adapter.id);
      
      // Perform async setup
      await setupExternalService();
      
      // Extend context
      return {
        context: {
          myService: new MyService(),
          myConfig: { /* ... */ }
        }
      };
    }
  };
};
```

The returned context is merged into the main auth context and is accessible in endpoints:

```typescript
const endpoint = createAuthEndpoint(
  "/my-endpoint",
  { method: "GET" },
  async (ctx) => {
    // Access plugin context
    const service = ctx.context.myService;
    return ctx.json(await service.getData());
  }
);
```

## Plugin Hooks

Plugins can register hooks that run before or after requests:

### Before Hooks

Run before the endpoint handler:

```typescript
hooks: {
  before: [
    {
      matcher: (ctx) => ctx.path === "/sign-up",
      handler: createAuthMiddleware(async (ctx) => {
        // Validate email domain
        const email = ctx.body?.email;
        if (!email?.endsWith("@company.com")) {
          throw APIError.from("BAD_REQUEST", {
            message: "Only company emails allowed"
          });
        }
      })
    }
  ]
}
```

### After Hooks

Run after the endpoint handler:

```typescript
hooks: {
  after: [
    {
      matcher: (ctx) => ctx.path === "/sign-up",
      handler: createAuthMiddleware(async (ctx) => {
        // Send welcome email
        const user = ctx.context.session?.user;
        if (user) {
          await sendWelcomeEmail(user.email);
        }
      })
    }
  ]
}
```

### Matchers

Matchers determine when hooks run:

```typescript
// Exact match
matcher: (ctx) => ctx.path === "/sign-in/email"

// Prefix match
matcher: (ctx) => ctx.path?.startsWith("/sign-in")

// Multiple paths
matcher: (ctx) => ['/sign-in/email', '/sign-up/email'].includes(ctx.path)

// All requests
matcher: () => true
```

## Plugin Schema

Plugins can define database tables:

```typescript
import type { BetterAuthPluginDBSchema } from "better-auth";

const schema = {
  // Table name
  myTable: {
    // Optional: Custom model name
    modelName: "my_custom_table",
    
    fields: {
      id: {
        type: "string",
        required: true,
        unique: true,
        defaultValue: () => generateId()
      },
      userId: {
        type: "string",
        required: true,
        references: {
          model: "user",
          field: "id"
        }
      },
      status: {
        type: "string",
        required: true,
        defaultValue: "pending"
      },
      metadata: {
        type: "string", // JSON string
        required: false
      },
      createdAt: {
        type: "date",
        required: true,
        defaultValue: () => new Date()
      }
    }
  }
} satisfies BetterAuthPluginDBSchema;
```

### Schema Customization

Users can customize plugin schemas:

```typescript
import { myPlugin } from "my-plugin";

const auth = betterAuth({
  plugins: [
    myPlugin({
      schema: {
        myTable: {
          modelName: "custom_table",
          fields: {
            // Rename fields
            status: "current_status"
          }
        }
      }
    })
  ]
});
```

## Plugin Endpoints

Plugins add endpoints using `createAuthEndpoint`:

```typescript
import { createAuthEndpoint } from "better-auth";
import * as z from "zod";

const endpoints = {
  myAction: createAuthEndpoint(
    "/my-action",
    {
      method: "POST",
      body: z.object({
        data: z.string()
      }),
      use: [sessionMiddleware], // Require authentication
      metadata: {
        openapi: {
          description: "Perform my action",
          responses: {
            200: {
              description: "Success",
              content: {
                "application/json": {
                  schema: {
                    type: "object",
                    properties: {
                      result: { type: "string" }
                    }
                  }
                }
              }
            }
          }
        }
      }
    },
    async (ctx) => {
      const { data } = ctx.body;
      const user = ctx.context.session.user;
      
      // Perform action
      const result = await performAction(user.id, data);
      
      return ctx.json({ result });
    }
  )
};
```

## Plugin Dependencies

Plugins can depend on other plugins:

```typescript
export const advancedPlugin = (options?: AdvancedOptions) => {
  return {
    id: "advanced-plugin",
    
    async init(ctx) {
      // Check if required plugin is installed
      const hasTwoFactor = ctx.hasPlugin("two-factor");
      
      if (!hasTwoFactor) {
        throw new Error("advanced-plugin requires two-factor plugin");
      }
      
      // Get plugin configuration
      const twoFactorPlugin = ctx.getPlugin("two-factor");
      
      return {};
    }
  };
};
```

## Type Safety

Plugins extend types automatically:

### Module Augmentation

```typescript
declare module "@better-auth/core" {
  interface BetterAuthPluginRegistry<AuthOptions, Options> {
    myPlugin: {
      creator: typeof myPlugin;
    };
  }
}
```

This enables:
- Type-safe plugin configuration
- Autocomplete in IDE
- Compile-time validation

### Inferring Plugin Types

```typescript
const auth = betterAuth({
  plugins: [twoFactor()]
});

// Types include two-factor fields
type Session = typeof auth.$Infer.Session;
// Session.user.twoFactorEnabled: boolean | undefined
```

## Plugin Best Practices

<AccordionGroup>
  <Accordion title="Use Descriptive IDs">
    Choose unique, descriptive plugin IDs:
    
    ```typescript
    // Good
    id: "two-factor-auth"
    
    // Bad
    id: "2fa" // too short
    id: "plugin" // not descriptive
    ```
  </Accordion>
  
  <Accordion title="Provide Sensible Defaults">
    Make plugins work out of the box:
    
    ```typescript
    export const myPlugin = (options?: MyOptions) => {
      const config = {
        enabled: true,
        timeout: 30000,
        ...options
      };
      
      return { /* plugin */ };
    };
    ```
  </Accordion>
  
  <Accordion title="Document Error Codes">
    Provide clear error codes and messages:
    
    ```typescript
    $ERROR_CODES: {
      INVALID_TOKEN: {
        code: "INVALID_TOKEN",
        message: "The provided token is invalid or expired"
      },
      MISSING_PERMISSION: {
        code: "MISSING_PERMISSION",
        message: "User does not have required permission"
      }
    }
    ```
  </Accordion>
  
  <Accordion title="Handle Cleanup">
    Clean up resources in hooks:
    
    ```typescript
    hooks: {
      after: [
        {
          matcher: (ctx) => ctx.path === "/sign-out",
          handler: async (ctx) => {
            // Clean up plugin-specific data
            await cleanupUserData(ctx.context.session?.user.id);
          }
        }
      ]
    }
    ```
  </Accordion>
  
  <Accordion title="Support Schema Customization">
    Allow users to customize table/field names:
    
    ```typescript
    export const myPlugin = (options?: MyOptions) => {
      return {
        schema: mergeSchema(defaultSchema, options?.schema)
      };
    };
    ```
  </Accordion>
</AccordionGroup>

```typescript
import { getTestInstance } from "better-auth/test";
import { myPlugin } from "./my-plugin";

describe("My Plugin", () => {
  it("should add endpoint", async () => {
    const { auth, client } = await getTestInstance({
      plugins: [myPlugin()]
    });
    
    const response = await client.myAction({ data: "test" });
    
    expect(response.data).toBeDefined();
  });
});
```
